#include <eracer.h>

#define PWM_Freq 38
#define PWM_Duty 25

'Initialise
dim TimeH(76)
dim TimeL(76)
dim Event(76)

OldStat = 0
NewStat = 0
DataCount = 0
InitTimer1(Osc, PS1_1/8)

'Wait for trigger
wait until button pressed
wait until button released

'Learn code
wait until IR_IN off
ClearTimer 1
StartTimer 1
NewStat = 255
set TL_LED on

do while TMR1H <= 250 and DataCount < 77
 OldStat = NewStat
 NewStat = 0
 if IR_IN off then NewStat = 1
 if NewStat <> OldStat then

  Temp = TMR1L
  Temp_H = TMR1H
  TimeH(DataCount) = Temp_H
  TimeL(DataCount) = Temp
  if NewStat = 1 then Event(DataCount) = TRUE
  if NewStat = 0 then Event(DataCount) = FALSE
  DataCount += 1
  
 end if
loop
set TL_LED off

'Write to EEPROM
For Temp = 1 to DataCount
 EPWrite Temp, TimeL(Temp)
 EPWrite Temp+80, TimeH(Temp)
 EPWrite Temp+160, Event(Temp)
next

'Playback
do
 Wait until button pressed
 wait until button released
 
 set TR_LED on
 ClearTimer 1
 for Temp = 1 to DataCount
  OldStat = TimeL(Temp)
  TempH = TimeH(Temp)
  wait while OldStat < TMR1L AND TempH < TMR1H
  OldStat = Event(Temp)
  if OldStat = TRUE then PWMOn
  if OldStat = FALSE then PWMOff  
 next
 set TR_LED off
loop
